{{- $fullName := include "graylog-auth-proxy.fullname" . -}}
{{- $svcPort := .Values.service.port -}}

Graylog Auth Proxy has been deployed!

{{- if .Values.ingress.enabled }}
Access the proxy via:
{{- range .Values.ingress.hosts }}
  https://{{ .host }}
{{- end }}
{{- else if eq .Values.service.type "LoadBalancer" }}
Access the proxy via the LoadBalancer IP:
  kubectl get svc {{ $fullName }}-svc -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}'
{{- else }}
Port-forward to access locally:
  kubectl port-forward svc/{{ $fullName }}-svc {{ $svcPort }}:{{ $svcPort }} -n {{ .Release.Namespace }}
  open https://localhost:{{ $svcPort }}
{{- end }}

Health check:
  kubectl exec -it deploy/{{ $fullName }} -n {{ .Release.Namespace }} -- wget -qO- --no-check-certificate https://localhost:{{ $svcPort }}/healthz

{{- if not .Values.metrics.serviceMonitor.enabled }}
NOTE: ServiceMonitor is not enabled. To enable Prometheus scraping:
  metrics.serviceMonitor.enabled=true
{{- end }}
